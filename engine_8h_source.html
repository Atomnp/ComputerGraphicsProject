546  void draw_bresenham_adjusted(int x1, int y1, int x2, int y2, const color &col = color(0)) {
547  int dx = abs(x2 - x1);
548  int dy = abs(y2 - y1);
549 
550  int lx = x2 > x1 ? 1 : -1;
551  int ly = y2 > y1 ? 1 : -1;
552 
553  int x = x1, y = y1;
554  bool changed = false;
555 
556  if (dx <= dy) {
557  changed = true;
558  std::swap(dx, dy);
559  std::swap(lx, ly);
560  std::swap(x, y);
561  }
562  int p = 2 * dy - dx;
563  for (int k = 0; k <= dx; k++) {
564  if (!changed)
565  putpixel_adjusted(x, y, 1, col);
566  else
567  putpixel_adjusted(y, x, 1, col);
568 
569  if (p < 0) {
570  x += lx;
571  p += 2 * dy;
572  } else {
573  x += lx;
574  y += ly;
575  p += 2 * dy - 2 * dx;
576  }
577  }

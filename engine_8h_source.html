337  inline color CalcDirLight(const vec3 &normal, const vec3 &viewdir, const color &col) {
338  auto diff = max(vec3::dot(normal, -dirlight.direction), 0.0f);

347  //function to calculate effect of a point light in a fragment with color col
348  color CalcPointLight(const pointLight &light, const vec3 &normal, const vec3 &fragPos, const vec3 &viewDir, const color &diffuseColor, float int_by_at) {
349  const vec3 lightDir = vec3::normalize(light.getpos() - fragPos);
350  const float diff = max(vec3::dot(normal, lightDir), 0.0);
351  const vec3 halfwayDir = vec3::normalize(lightDir + viewDir);
352  const float spec = pow(max(vec3::dot(normal, halfwayDir), 0.0), currentMaterial->shininess);
353  const color ambient = diffuseColor * light.get_ambient_color() * (currentMaterial->AmbientStrength * int_by_at);
354  const color diffuse = diffuseColor * light.get_diffuse_color() * (currentMaterial->DiffuseStrength * diff * int_by_at);
355  const color specular = light.get_diffuse_color() * (currentMaterial->SpecularStrength * spec * int_by_at);
356  return (ambient + diffuse + specular);

618  float dist = vec3::dist(extraInfoAboutVertex, light.getpos());
619  float int_by_at = light.intensity / (light.constant + light.linear * dist + light.quadratic * (dist * dist));

626  extraInfoAboutVertex += vec3(currentMaterial->AmbientStrength * int_by_at, currentMaterial->DiffuseStrength * diff * int_by_at, currentMaterial->SpecularStrength * spec * int_by_at);

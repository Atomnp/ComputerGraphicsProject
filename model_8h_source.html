72 static drawable<Vertex> *processMesh(aiMesh *mesh, const aiScene *scene) {
73  // data to fill
74  std::vector<Vertex> vertices;

76  std::vector<std::string> textures;
77 
78  // walk through each of the mesh's vertices
79  for (unsigned int i = 0; i < mesh->mNumVertices; i++) {
80  Vertex vertex;
81  glm::vec3 vector; // we declare a placeholder vector since assimp uses its own vector class that doesn't directly convert to glm's vec3 class so we transfer the data to this placeholder glm::vec3 first.
82  // positions
83  vector.x = mesh->mVertices[i].x;
84  vector.y = mesh->mVertices[i].y;
85  vector.z = mesh->mVertices[i].z;
86  vertex.position = vector;
87  // normals
88  if (mesh->HasNormals()) {
89  vector.x = mesh->mNormals[i].x;

91  vector.z = mesh->mNormals[i].z;
92  vertex.normal = vector;
93  }
94  // texture coordinates
95  if (mesh->mTextureCoords[0]) // does the mesh contain texture coordinates?
96  {
97  glm::vec2 vec;
98  // a vertex can contain up to 8 different texture coordinates. We thus make the assumption that we won't
99  // use models where a vertex can have multiple texture coordinates so we always take the first set (0).
100  vec.x = mesh->mTextureCoords[0][i].x;
101  vec.y = mesh->mTextureCoords[0][i].y;
102  vertex.texCoord = vec;
103  // tangent
104  vector.x = mesh->mTangents[i].x;
105  vector.y = mesh->mTangents[i].y;
106  vector.z = mesh->mTangents[i].z;
107  vertex.tangent = vector;
108  // bitangent
109  vector.x = mesh->mBitangents[i].x;
110  vector.y = mesh->mBitangents[i].y;
111  vector.z = mesh->mBitangents[i].z;
112  vertex.bitangent = vector;
113  } else
114  vertex.texCoord = glm::vec2(0.0f, 0.0f);
115 
116  vertices.push_back(vertex);
117  }
118 
119  // now wak through each of the mesh's faces (a face is a mesh its triangle) and retrieve the corresponding vertex indices.
120  for (unsigned int i = 0; i < mesh->mNumFaces; i++) {
121  aiFace face = mesh->mFaces[i];
122  // retrieve all indices of the face and store them in the indices vector
123  for (unsigned int j = 0; j < face.mNumIndices; j++)
124  indices.push_back(face.mIndices[j]);
125  }
126  // process materials
127  aiMaterial *material = scene->mMaterials[mesh->mMaterialIndex];
128 
129  // 0. ambient maps
130  std::string ambientMaps = loadMaterialTexture(material, aiTextureType_AMBIENT, "texture_ambient");
131  textures.push_back(ambientMaps);
132  // 1. diffuse maps
133  std::string diffuseMaps = loadMaterialTexture(material, aiTextureType_DIFFUSE, "texture_diffuse");
134  textures.push_back(diffuseMaps);
135  // 2. specular maps
136  std::string specularMaps = loadMaterialTexture(material, aiTextureType_SPECULAR, "texture_specular");
137  textures.push_back(specularMaps);
138  // 3. normal maps
139  std::string normalMaps = loadMaterialTexture(material, aiTextureType_HEIGHT, "texture_normal");
140  textures.push_back(normalMaps);
141 
142  std::string name(mesh->mName.C_Str());
143 
144  while (meshes_loaded.find(name) != meshes_loaded.end()) {
145  name += "_copy";
146  }
147  meshes_loaded[name] = Mesh(vertices, indices, name, textures);
148  Mesh &m = meshes_loaded[name];
149 
150  aiColor4D diffuse, specular;
151  float shininess = 0;
152  m.material.diffuseColor = AI_SUCCESS != aiGetMaterialColor(material, AI_MATKEY_COLOR_DIFFUSE, &diffuse) ? vec3(1, 0, 0) : vec3(diffuse.r, diffuse.g, diffuse.b);
153  m.material.specularColor = AI_SUCCESS != aiGetMaterialColor(material, AI_MATKEY_COLOR_SPECULAR, &specular) ? vec3(1, 1, 1) : vec3(specular.r, specular.g, specular.b);
154  m.material.shininess = AI_SUCCESS != aiGetMaterialFloat(material, AI_MATKEY_SHININESS, &shininess) ? 512 : shininess;
155 
156  return &m;
157 }

160 static void processNode(aiNode *node, const aiScene *scene, std::vector<drawable<Vertex> *> &meshes, int num) {
161  int temp = num;

168  aiMesh *mesh = scene->mMeshes[node->mMeshes[i]];

175  processNode(node->mChildren[i], scene, meshes, ++num);

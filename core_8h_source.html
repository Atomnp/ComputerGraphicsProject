63inline bool GLLogCall(const char *function, const char *file, int line) {
64 while (GLenum errorcode = glGetError()) {
65 std::string error;
66 switch (errorcode) {
67 case GL_INVALID_ENUM:
68 error = "INVALID_ENUM";
69 break;
70 case GL_INVALID_VALUE:
71 error = "INVALID_VALUE";
72 break;
73 case GL_INVALID_OPERATION:
74 error = "INVALID_OPERATION";
75 break;
76 case GL_STACK_OVERFLOW:
77 error = "STACK_OVERFLOW";
78 break;
79 case GL_STACK_UNDERFLOW:
80 error = "STACK_UNDERFLOW";
81 break;
82 case GL_OUT_OF_MEMORY:
83 error = "OUT_OF_MEMORY";
84 break;
85 case GL_INVALID_FRAMEBUFFER_OPERATION:
86 error = "INVALID_FRAMEBUFFER_OPERATION";
87 break;
88 }
89 std::cout << "[Opengl Error] ( Error code :" << errorcode << " : " << error << " )" << function << " " << file << " : " << line << std::endl;

92 return true;
93}

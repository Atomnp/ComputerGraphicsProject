118 typedef std::common_type_t<T1, T2> T;
119 T the_first_index = (T)first_index;
120 T last_index = (T)index_after_last;
121 if (the_first_index == last_index)
122 return;
123 if (last_index < the_first_index) {
124 T temp = last_index;
125 last_index = the_first_index;
126 the_first_index = temp;
127 }
128 last_index--;
129 if (num_blocks == 0)
130 num_blocks = thread_count;
131 ui64 total_size = (ui64)(last_index - the_first_index + 1);
132 ui64 block_size = (ui64)(total_size / num_blocks);
133 if (block_size == 0) {
134 block_size = 1;
135 num_blocks = (ui32)total_size > 1 ? (ui32)total_size : 1;
136 }
137 std::atomic<ui32> blocks_running = 0;
138 for (ui32 t = 0; t < num_blocks; t++) {
139 T start = ((T)(t * block_size) + the_first_index);
140 T end = (t == num_blocks - 1) ? last_index + 1 : ((T)((t + 1) * block_size) + the_first_index);

145 });

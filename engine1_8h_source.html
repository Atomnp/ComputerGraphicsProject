540 inline color CalcDirLight(const vec3 &normal, const vec3 &viewdir, const color &col, const Material *material) const {
541 auto diff = max(vec3::dot(normal, -dirlight.direction), 0.0f);
542 auto spec = pow(max(vec3::dot(viewdir, reflect(dirlight.direction, normal)), 0.0), material->shininess);
543 color diffuse = col * dirlight.col * dirlight.intensity * material->DiffuseStrength * diff;
544 color specular = dirlight.col * dirlight.intensity * material->SpecularStrength * spec;
545 diffuse += specular;
546 diffuse.a() = 255;
547 return std::move(diffuse);
548 }

553 static color CalcPointLight(const pointLight &light, const vec3 &normal, const vec3 &fragPos, const vec3 &viewDir, const color &diffuseColor, float int_by_at, const Material *material) {
554 const vec3 lightDir = vec3::normalize(light.getpos() - fragPos);
555 const float diff = max(vec3::dot(normal, lightDir), 0.0);
556 const vec3 halfwayDir = vec3::normalize(lightDir + viewDir);
557 const float spec = pow(max(vec3::dot(normal, halfwayDir), 0.0), material->shininess);
558 const color ambient = diffuseColor * light.get_ambient_color() * (material->AmbientStrength * int_by_at);
559 const color diffuse = diffuseColor * light.get_diffuse_color() * (material->DiffuseStrength * diff * int_by_at);
560 const color specular = light.get_diffuse_color() * (material->SpecularStrength * spec * int_by_at);
561 // return (ambient + diffuse + specular);
562 return (diffuse + specular);
563 }

594 void draw_bresenham_adjusted(int x1, int y1, int x2, int y2, const color &col = color(0)) {
595 int dx = abs(x2 - x1);
596 int dy = abs(y2 - y1);
597
598 int lx = x2 > x1 ? 1 : -1;
599 int ly = y2 > y1 ? 1 : -1;
600
601 int x = x1, y = y1;
602 bool changed = false;
603
604 if (dx <= dy) {
605 changed = true;
606 std::swap(dx, dy);
607 std::swap(lx, ly);
608 std::swap(x, y);
609 }
610 int p = 2 * dy - dx;
611 for (int k = 0; k <= dx; k++) {
612 if (!changed)
613 putpixel_adjusted(x, y, 1, col);
614 else
615 putpixel_adjusted(y, x, 1, col);
616
617 if (p < 0) {
618 x += lx;
619 p += 2 * dy;
620 } else {
621 x += lx;
622 y += ly;
623 p += 2 * dy - 2 * dx;
624 }
625 }

655 void fillBottomFlatTriangle(const Vertex2 &v0, const Vertex2 &v1, const Vertex2 &v2, Mesh *mesh) {
656 assert(fabs(v1.v.position.y - v0.v.position.y) < 0.001);
657 assert(v1.v.position.x >= v0.v.position.x);
658
659 // calulcate Inverse slope in screen space
660 float m0 = (v2.v.position.x - v0.v.position.x) / (v2.v.position.y - v0.v.position.y);
661 float m1 = (v2.v.position.x - v1.v.position.x) / (v2.v.position.y - v1.v.position.y);
662
663 // calculate start and end scanlines
664 int yStart = (int)ceil(v0.v.position.y - 0.5f);
665 const int yEnd = (int)ceil(v2.v.position.y - 0.5f); // the scanline AFTER the last line drawn
666
667 const float unit0 = 1.f / (v2.v.position.y - v0.v.position.y), unit1 = 1.f / (v2.v.position.y - v1.v.position.y);
668 float u0 = 0, u1 = 0;
669
670 Vertex2 diff0 = v2 - v0, diff1 = v2 - v1, diff2;
671 Vertex2 vx0, vx1, vx2;
672 for (int y = yStart; y < yEnd && y < fboCPU->ymax; ++y, u0 += unit0, u1 += unit1) {
673 if (y <= fboCPU->ymin) {
674 continue;
675 }
676 // assert(u0 <= 1 && u1 <= 1);
677 // caluclate start and end points (x-coords)
678 // add 0.5 to y value because we're calculating based on pixel CENTERS
679 const float px0 = m0 * (float(y) + 0.5f - v0.v.position.y) + v0.v.position.x;
680 const float px1 = m1 * (float(y) + 0.5f - v1.v.position.y) + v1.v.position.x;
681
682 vx0 = v0 + diff0 * u0;
683 vx1 = v1 + diff1 * u1;
684
685 // calculate start and end pixels
686 int xStart = (int)ceil(px0 - 0.5f);
687 const int xEnd = (int)ceil(px1 - 0.5f); // the pixel AFTER the last pixel drawn
688
689 const float unit2 = 1.f / (xEnd - xStart);
690 float u2 = 0;
691 diff2 = vx1 - vx0;
692
693 for (int x = xStart; x < xEnd && x < fboCPU->xmax; x++, u2 += unit2) {
694 if (x <= fboCPU->xmin) {
695 continue;
696 }
697 float z = 1 / (vx0.v.position.w + (diff2.v.position.w) * u2);
698#ifdef MULTITHREADED
699 const std::lock_guard<std::mutex> lock(fboCPUMutex);
700#endif
701 auto gotz = getpixelZ_adjusted(x, y);
702 if (gotz > z) {
703 vx2 = vx0 + diff2 * u2;
704 vx2 /= vx2.v.position.w;
705 putpixel_adjusted(x, y, z, getcolor(vx2, mesh));
